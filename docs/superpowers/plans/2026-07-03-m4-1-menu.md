# M4.1 动态路由菜单实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现 GET /api/system/menus，下发完整菜单树（MenuDTO 含 meta.permissions），前端负责按角色过滤。

**Architecture:** model.Menu 加 Title（显示名，Name 为路由名）。后端查全部菜单→构建树→转 MenuDTO（PermissionCode 转 meta.permissions.any）→下发。不过滤（前端 canSeeMenu + 路由守卫过滤）。复用 M3 中间件。

**Tech Stack:** Go 1.25、Gin v1.12、GORM v1.31、SQLite 测试、testify。

**Spec:** `docs/superpowers/specs/2026-07-03-m4-1-menu-design.md`

// seedMenus 种入菜单。根菜单先建（拿到 ID），子菜单 ParentID 指向根。
func (s *AuthService) seedMenus(ctx context.Context) error {
	// 先种根菜单（ParentID=0），建 name→id 映射
	rootIDs := make(map[string]uint)
	for _, m := range seedMenus {
		if m.ParentID != 0 || m.Name != "system" {
			// 只先种 system 根（其他根 home/crud 无子菜单，直接种）
		}
	}
	// 简化：先种所有 ParentID=0 的根，再种子
	for i := range seedMenus {
		if seedMenus[i].ParentID == 0 && seedMenus[i].Name != "system" {
			// home/crud 根
			m := seedMenus[i]
			if err := s.db.WithContext(ctx).Where(model.Menu{Name: m.Name}).FirstOrCreate(&m).Error; err != nil {
				return err
			}
		}
	}
	// system 根
	var systemMenu model.Menu
	if err := s.db.WithContext(ctx).Where(model.Menu{Name: "system"}).FirstOrCreate(&systemMenu, model.Menu{Name: "system", Title: "系统管理", Path: "/system", Icon: "Setting", Sort: 20, ShowMenu: true}).Error; err != nil {
		return err
	}
	rootIDs["system"] = systemMenu.ID
	// 子菜单（systemUser/Role/Permission）ParentID = systemMenu.ID
	for _, m := range seedMenus {
		if m.PermissionCode != "" {
			m.ParentID = systemMenu.ID
			existing := model.Menu{}
			if err := s.db.WithContext(ctx).Where(model.Menu{Name: m.Name}).FirstOrCreate(&existing, m).Error; err != nil {
				return err
			}
		}
	}
	return nil
}
```
> 注：seedMenus 实现较繁，因 ParentID 依赖 system 的 ID。实现时确保幂等（FirstOrCreate by Name）。

package repository

import (
	"context"

	"gorm.io/gorm"
	"gva/internal/model"
)

// MenuRepository 菜单数据访问。M4.1 只需 GetAllMenus。
type MenuRepository interface {
	GetAllMenus(ctx context.Context) ([]model.Menu, error)
}

type menuRepository struct {
	db *gorm.DB
}

func NewMenuRepository(db *gorm.DB) MenuRepository {
	return &menuRepository{db: db}
}

// GetAllMenus 查全部菜单，按 parent_id, sort 排序。
func (r *menuRepository) GetAllMenus(ctx context.Context) ([]model.Menu, error) {
	var menus []model.Menu
	err := r.db.WithContext(ctx).Order("parent_id, sort").Find(&menus).Error
	return menus, err
}

package service

import (
	"context"

	"gva/internal/model"
	"gva/internal/repository"
)

// MenuDTO 对齐前端 lib/router/types-menu.ts。
type MenuDTO struct {
	Path      string    `json:"path"`
	Name      string    `json:"name"`
	Component string    `json:"component,omitempty"`
	Meta      MenuMeta  `json:"meta"`
	Children  []MenuDTO `json:"children,omitempty"`
}

// MenuMeta 菜单元信息。Permissions 非空时前端按权限过滤。
type MenuMeta struct {
	Title       string           `json:"title"`
	Icon        string           `json:"icon,omitempty"`
	ShowMenu    bool             `json:"showMenu"`
	Permissions *MenuPermissions `json:"permissions,omitempty"`
}

// MenuPermissions 权限约束。Any=任一命中即可见，All=全部命中才可见。
type MenuPermissions struct {
	Any []string `json:"any,omitempty"`
	All []string `json:"all,omitempty"`
}

// MenuService 菜单业务：构建树 + 转 DTO。
type MenuService struct {
	repo repository.MenuRepository
}

func NewMenuService(repo repository.MenuRepository) *MenuService {
	return &MenuService{repo: repo}
}

// GetMenus 查全部菜单，构建树，转 MenuDTO。后端下发完整树，前端过滤。
func (s *MenuService) GetMenus(ctx context.Context) ([]MenuDTO, error) {
	menus, err := s.repo.GetAllMenus(ctx)
	if err != nil {
		return nil, err
	}
	return buildMenuTree(menus), nil
}

// buildMenuTree 扁平菜单构建为树（按 ParentID 递归）。
func buildMenuTree(menus []model.Menu) []MenuDTO {
	byID := make(map[uint]*MenuDTO, len(menus))
	order := make([]uint, 0, len(menus))
	for i := range menus {
		m := &menus[i]
		dto := toMenuDTO(m)
		byID[m.ID] = &dto
		order = append(order, m.ID)
	}
	roots := make([]MenuDTO, 0)
	for _, id := range order {
		dto := byID[id]
		menu := findMenuByID(menus, id)
		if menu.ParentID == 0 || byID[menu.ParentID] == nil {
			roots = append(roots, *dto)
		} else {
			parent := byID[menu.ParentID]
			parent.Children = append(parent.Children, *dto)
		}
	}
	return roots
}

// findMenuByID 在扁平列表找原始 Menu（取 ParentID）。
func findMenuByID(menus []model.Menu, id uint) *model.Menu {
	for i := range menus {
		if menus[i].ID == id {
			return &menus[i]
		}
	}
	return nil
}

// toMenuDTO 单个 Menu→DTO。PermissionCode 非空转 meta.permissions.any。
func toMenuDTO(m *model.Menu) MenuDTO {
	dto := MenuDTO{
		Path:      m.Path,
		Name:      m.Name,
		Component: m.Component,
		Meta: MenuMeta{
			Title:    m.Title,
			Icon:     m.Icon,
			ShowMenu: m.ShowMenu,
		},
	}
	if m.PermissionCode != "" {
		dto.Meta.Permissions = &MenuPermissions{Any: []string{m.PermissionCode}}
	}
	return dto
}
```
